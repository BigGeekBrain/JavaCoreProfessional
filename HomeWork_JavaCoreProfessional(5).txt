public class Car implements Runnable {
    
    private static CountDownLatch cdLOnFinish = new CountDownLatch(4);

    public static CountDownLatch getCdLOnFinish() {
        return cdLOnFinish;
    }

    @Override
    public void run() {
        try {
            System.out.println(this.name + " ãîòîâèòñÿ");
            Thread.sleep(500 + (int)(Math.random() * 800));
            System.out.println(this.name + " ãîòîâ");
            MainClass.prepare.countDown();
            MainClass.prepare.await();

        } catch (Exception e) {
            e.printStackTrace();
        }

        for (int i = 0; i < race.getStages().size(); i++) {
            race.getStages().get(i).go(this);
        }
        cdLOnFinish.countDown();
    }
}

public class Tunnel extends Stage {

    private Semaphore semaphore;

    public Tunnel() {
        this.length = 80;
        this.description = "Òîííåëü " + length + " ìåòðîâ";
        this.semaphore = new Semaphore(2);
    }

    @Override
    public void go(Car c) {
        try {
            try {
                System.out.println(c.getName() + " ãîòîâèòñÿ ê ýòàïó(æäåò): " + description);

                if (!semaphore.tryAcquire()) {
                    System.out.println(c.getName() + " íà÷àë ýòàï: " + description);
                    Thread.sleep(length / c.getSpeed() * 1000);
                }

            } catch (InterruptedException e) {
                e.printStackTrace();
            } finally {
                System.out.println(c.getName() + " çàêîí÷èë ýòàï: " + description);
                semaphore.release();
            }
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
